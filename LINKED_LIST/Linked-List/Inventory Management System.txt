class ItemNode {
    int id;
    String name;
    int quantity;
    double price;
    ItemNode next;

    ItemNode(int i,String n,int q,double p) {
        id = i;
        name = n;
        quantity = q;
        price = p;
        next = null;
    }
}

class InventoryList {

    ItemNode head = null;

    void insertBeginning(int i,String n,int q,double p) {

        ItemNode newNode = new ItemNode(i,n,q,p);
        newNode.next = head;
        head = newNode;
    }

    void insertEnd(int i,String n,int q,double p) {

        ItemNode newNode = new ItemNode(i,n,q,p);

        if(head == null) {
            head = newNode;
            return;
        }

        ItemNode temp = head;

        while(temp.next != null) {
            temp = temp.next;
        }

        temp.next = newNode;
    }

    void insertPosition(int i,String n,int q,double p,int pos) {

        if(pos == 1) {
            insertBeginning(i,n,q,p);
            return;
        }

        ItemNode newNode = new ItemNode(i,n,q,p);
        ItemNode temp = head;

        for(int j=1;j<pos-1 && temp!=null;j++) {
            temp = temp.next;
        }

        if(temp == null)
            return;

        newNode.next = temp.next;
        temp.next = newNode;
    }

    void deleteItem(int id) {

        if(head == null)
            return;

        if(head.id == id) {
            head = head.next;
            return;
        }

        ItemNode temp = head;

        while(temp.next != null && temp.next.id != id) {
            temp = temp.next;
        }

        if(temp.next != null) {
            temp.next = temp.next.next;
        }
    }

    void updateQuantity(int id,int newQty) {

        ItemNode temp = head;

        while(temp != null) {

            if(temp.id == id) {
                temp.quantity = newQty;
                return;
            }

            temp = temp.next;
        }
    }

    void searchById(int id) {

        ItemNode temp = head;

        while(temp != null) {

            if(temp.id == id) {
                System.out.println(temp.id+" "+temp.name+" "+temp.quantity+" "+temp.price);
                return;
            }

            temp = temp.next;
        }

        System.out.println("Item not found");
    }

    void searchByName(String name) {

        ItemNode temp = head;

        while(temp != null) {

            if(temp.name.equals(name)) {
                System.out.println(temp.id+" "+temp.name+" "+temp.quantity+" "+temp.price);
                return;
            }

            temp = temp.next;
        }

        System.out.println("Item not found");
    }

    void totalValue() {

        double total = 0;
        ItemNode temp = head;

        while(temp != null) {
            total += temp.quantity * temp.price;
            temp = temp.next;
        }

        System.out.println("Total Inventory Value: "+total);
    }

    void sortByNameAsc() {

        for(ItemNode i=head;i!=null;i=i.next) {
            for(ItemNode j=i.next;j!=null;j=j.next) {

                if(i.name.compareTo(j.name) > 0) {

                    int id=i.id; String name=i.name; int q=i.quantity; double p=i.price;

                    i.id=j.id; i.name=j.name; i.quantity=j.quantity; i.price=j.price;

                    j.id=id; j.name=name; j.quantity=q; j.price=p;
                }
            }
        }
    }

    void sortByPriceDesc() {

        for(ItemNode i=head;i!=null;i=i.next) {
            for(ItemNode j=i.next;j!=null;j=j.next) {

                if(i.price < j.price) {

                    int id=i.id; String name=i.name; int q=i.quantity; double p=i.price;

                    i.id=j.id; i.name=j.name; i.quantity=j.quantity; i.price=j.price;

                    j.id=id; j.name=name; j.quantity=q; j.price=p;
                }
            }
        }
    }

    void display() {

        ItemNode temp = head;

        while(temp != null) {
            System.out.println(temp.id+" "+temp.name+" "+temp.quantity+" "+temp.price);
            temp = temp.next;
        }
    }
}

public class Main {

    public static void main(String[] args) {

        InventoryList list = new InventoryList();

        list.insertBeginning(1,"Keyboard",10,500);
        list.insertEnd(2,"Mouse",15,300);
        list.insertEnd(3,"Monitor",5,8000);

        System.out.println("Inventory");
        list.display();

        list.updateQuantity(2,20);

        System.out.println("\nSearch Item");
        list.searchById(2);

        list.totalValue();

        list.sortByNameAsc();
        System.out.println("\nSorted by Name");
        list.display();

        list.sortByPriceDesc();
        System.out.println("\nSorted by Price Desc");
        list.display();

        list.deleteItem(1);
        System.out.println("\nAfter Deletion");
        list.display();
    }
}
